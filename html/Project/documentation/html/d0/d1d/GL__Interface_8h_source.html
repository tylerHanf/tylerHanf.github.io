100 bool AddData(GeomData& data, Handle& handle)
101 {
102 // Find a set that matches this data format and has enough room
103 for (int i = 0; i < buffs.size(); ++i) {
104 BuffSet& buff = buffs[i];
105 if (buff.attr == data.attr) {
106 if (data.VertSizeBytes() < buff.vbo.NumBytesUnused()) {
107 if (!data.hasIndices ||
108 (data.hasIndices &&
109 data.IndicesSizeBytes() < buff.ebo.NumBytesUnused())) {
110 handle.bHandle = i;
111 break;
112 }
113 }
114 }
115 }
116
117 // Check if we need to make a new buffer set
118 if (handle.bHandle < 0) {
119 handle.bHandle = MakeSet(data.attr);
120 }
121
122 BuffSet& set = buffs[handle.bHandle];
123 int offset = set.vbo.Add(data.GetVertPtr(), data.NumVerts());
124 if (offset < 0) {
125 Logger::Log("Failed to add to VBO");
126 return false;
127 }
128
129 handle.offset = offset;
130
131 if (data.hasIndices) {
132 if (set.ebo.Add(
133 data.GetIndicesPtr(), data.NumIndices(), sizeof(unsigned)) < 0) {
134 Logger::Log("Failed to add to EBO");
135 return false;
136 }
137 }
138
139 return true;
140 }
