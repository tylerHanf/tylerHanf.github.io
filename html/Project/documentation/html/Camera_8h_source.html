272 // Make view transform

275
276 // Calculate pixel sizes
277 pixelWidth = width / imageWidth;
278 pixelHeight = height / imageHeight;
279
280 top_left_cam = Vec3(
281 0.5 * (-width + pixelWidth), 0.5 * (height - pixelHeight), -1 / near);
282 }

293 {
294 Vec3 pixPos(top_left_cam.x + pixelWidth * s,
295 top_left_cam.y - pixelHeight * t,
296 top_left_cam.z);
297 Vec3 pixPos_world = (viewInverse * pixPos).Row(3);
298
299 return Ray(eye, pixPos_world);
300 }
