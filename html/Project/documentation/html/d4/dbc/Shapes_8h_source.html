24MakeSphere(unsigned stacks, unsigned slices)

26 GeomData sphere;

28 // loop through stacks.
29 for (int i = 0; i <= stacks; ++i) {
30
31 float V = (float)i / (float)stacks;
32 float phi = V * PI;
33
34 // loop through the slices.
35 for (int j = 0; j <= slices; ++j) {
36
37 float U = (float)j / (float)slices;

46 sphere.verts.push_back(y);

57 float tv = asin(normal.y) / PI + 0.5;
58 sphere.verts.push_back(tu);
59 sphere.verts.push_back(tv);
60 }
61 }

65 sphere.indices.push_back(i);
